## leetcode 1476 : subrectangle queries

```c++
class Matrix{
	public:
	Matrix(vector<vector<int>> _m):m(_m){}
	vector<vector<int>> m;
	void print_matrix();
}

 ```


```c++
void print_matrix(){
	for(auto it_i = m.begin(); it_i != m.end(); ++ it_i) {
        for (int j = 0; j <= 2; j++)
            cout << (*it_i)[j] << " ";
        cout << endl;
    }
}
 ```


```c++
void print_matrix(){
	for(int i = 0; i <= 2; i ++) {
        for (auto it_j = m[i].begin(); it_j != m[i].end(); ++ it_j)
            cout << *it_j << " ";
        cout << endl;
    }
}
 ```



```c++
void print_matrix(){

}
 ```
